Docker commands
# 1. Detener y eliminar todos los contenedores
docker-compose down -v

# 2. Eliminar contenedores huérfanos manualmente
docker container prune -f

# 3. Eliminar las imágenes viejas
docker rmi poems_node1 poems_node2 poems_node3 poems_node4 poems_node5 poems_node6 poems_node7 poems_node8 2>/dev/null || true

# 4. Ahora reconstruir desde cero
docker-compose build --no-cache

# 5. Levantar los contenedores
docker-compose up
docker-compose --build
